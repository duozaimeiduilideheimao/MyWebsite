2.页面乱码,

2.添加init-parm,在如下标签中添加

default org.apache.catalina.servlets.DefaultServlet debug 0 listings false 1 fileEncoding UTF-8 //添加utf编码 3.配置tomcat/bin/catalina.bat中的

set “JAVA_OPTS=%JAVA_OPTS% %JSSE_OPTS%”
